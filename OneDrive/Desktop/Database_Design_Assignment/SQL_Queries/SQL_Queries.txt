##Table Creation
#Create Students Table:

CREATE TABLE Students (
    StudentID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    Address VARCHAR2(255),
    Phone VARCHAR2(15),
    Email VARCHAR2(100),
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Courses Table:

CREATE TABLE Courses (
    CourseID NUMBER PRIMARY KEY,
    Title VARCHAR2(100),
    Credits NUMBER,
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Enrolments Table:

CREATE TABLE Enrolments (
    EnrolmentID NUMBER PRIMARY KEY,
    StudentID NUMBER,
    CourseID NUMBER,
    SemesterID NUMBER,
    Grade CHAR(2),
    FOREIGN KEY (StudentID) REFERENCES Students(StudentID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (SemesterID) REFERENCES Semesters(SemesterID),
    CONSTRAINT unique_enrolment UNIQUE (StudentID, CourseID, SemesterID)
);


#Create Instructors Table:

CREATE TABLE Instructors (
    InstructorID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Departments Table:

CREATE TABLE Departments (
    DepartmentID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    HeadOfDepartment VARCHAR2(100)
);


#Create Semesters Table:	

CREATE TABLE Semesters (
    SemesterID NUMBER PRIMARY KEY,
    StartDate DATE,
    EndDate DATE
);


#Create CourseInstructors Table:

CREATE TABLE CourseInstructors (
    CourseID NUMBER,
    InstructorID NUMBER,
    PRIMARY KEY (CourseID, InstructorID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (InstructorID) REFERENCES Instructors(InstructorID)
);

#Create Prerequisites Table
CREATE TABLE Prerequisites (
    CourseID NUMBER,
    PrerequisiteID NUMBER,
    PRIMARY KEY (CourseID, PrerequisiteID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (PrerequisiteID) REFERENCES Courses(CourseID)
);

#Create Sections Table
CREATE TABLE Sections (
    SectionID NUMBER PRIMARY KEY,
    CourseID NUMBER,
    SemesterID NUMBER,
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (SemesterID) REFERENCES Semesters(SemesterID)
);

-- Checking for unique student IDs
SELECT StudentID, COUNT(*) AS NumOccurrences
FROM Students
GROUP BY StudentID
HAVING COUNT(*) > 1;

-- Checking for students with updated addresses
SELECT StudentID, Address
FROM Students
WHERE Address <> 'Initial_Address';

-- Checking for students with updated phone numbers
SELECT StudentID, Phone
FROM Students
WHERE Phone <> 'Initial_Phone';

-- Checking for students with updated email addresses
SELECT StudentID, Email
FROM Students
WHERE Email <> 'Initial_Email';

-- Listing courses with multiple sections
SELECT CourseID, COUNT(*) AS NumSections
FROM Sections
GROUP BY CourseID
HAVING COUNT(*) > 1;

-- Listing instructors teaching multiple courses
SELECT InstructorID, COUNT(DISTINCT CourseID) AS NumCourses
FROM CourseInstructors
GROUP BY InstructorID
HAVING COUNT(DISTINCT CourseID) > 1;

-- Create table Prerequisites
CREATE TABLE Prerequisites (
  CourseID NUMBER PRIMARY KEY,
  PrerequisiteID NUMBER,
  FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
  FOREIGN KEY (PrerequisiteID) REFERENCES Courses(CourseID)
);

-- Listing courses and their prerequisites
SELECT c.Title AS Course, c.Credits, pc.Title AS Prerequisite
FROM Courses c
LEFT JOIN Prerequisites p ON c.CourseID = p.CourseID
LEFT JOIN Courses pc ON p.PrerequisiteID = pc.CourseID;

-- Check for duplicate enrolments by student
SELECT StudentID, CourseID, SemesterID, COUNT(*) AS NumEnrolments
FROM Enrolments
GROUP BY StudentID, CourseID, SemesterID
HAVING COUNT(*) > 1;

-- List enrolment records with grades
SELECT StudentID, CourseID, SemesterID, Grade
FROM Enrolments
WHERE Grade IS NOT NULL;


-- Check for unique instructor IDs:
SELECT InstructorID, COUNT(*) AS NumOccurrences
FROM Instructors
GROUP BY InstructorID
HAVING COUNT(*) > 1;

-- List instructors and their teaching assignments
SELECT i.InstructorID, i.Name, c.Title AS Course
FROM Instructors i
INNER JOIN CourseInstructors ci ON i.InstructorID = ci.InstructorID
INNER JOIN Courses c ON ci.CourseID = c.CourseID;
