Q.3 - Joining Two Array Lists
Write a Java program to join two Array Lists.

package Q1;
import java.util.ArrayList;
import java.util.List;

public class join {
    public static void main(String[] args) {
        // Define two ArrayLists
        List<Integer> list1 = new ArrayList<>();
        list1.add(1);
        list1.add(2);
        list1.add(3);

        List<Integer> list2 = new ArrayList<>();
        list2.add(4);
        list2.add(5);
        list2.add(6);

        // Join the two ArrayLists
        List<Integer> combinedList = joinArrayLists(list1, list2);

        // Display the resulting combined ArrayList
        System.out.println("Combined ArrayList: " + combinedList);
    }

    // Function to join two ArrayLists
    private static <T> List<T> joinArrayLists(List<T> list1, List<T> list2) {
        List<T> combinedList = new ArrayList<>(list1);
        combinedList.addAll(list2);
        return combinedList;
    }
}
