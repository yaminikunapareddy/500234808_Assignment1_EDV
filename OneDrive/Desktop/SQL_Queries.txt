##Table Creation
#Create Students Table:

CREATE TABLE Students (
    StudentID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    Address VARCHAR2(255),
    Phone VARCHAR2(15),
    Email VARCHAR2(100),
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Courses Table:

CREATE TABLE Courses (
    CourseID NUMBER PRIMARY KEY,
    Title VARCHAR2(100),
    Credits NUMBER,
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Enrolments Table:

CREATE TABLE Enrolments (
    EnrolmentID NUMBER PRIMARY KEY,
    StudentID NUMBER,
    CourseID NUMBER,
    SemesterID NUMBER,
    Grade CHAR(2),
    FOREIGN KEY (StudentID) REFERENCES Students(StudentID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (SemesterID) REFERENCES Semesters(SemesterID),
    CONSTRAINT unique_enrolment UNIQUE (StudentID, CourseID, SemesterID)
);


#Create Instructors Table:

CREATE TABLE Instructors (
    InstructorID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    DepartmentID NUMBER,
    FOREIGN KEY (DepartmentID) REFERENCES Departments(DepartmentID)
);


#Create Departments Table:

CREATE TABLE Departments (
    DepartmentID NUMBER PRIMARY KEY,
    Name VARCHAR2(100),
    HeadOfDepartment VARCHAR2(100)
);


#Create Semesters Table:	

CREATE TABLE Semesters (
    SemesterID NUMBER PRIMARY KEY,
    StartDate DATE,
    EndDate DATE
);


#Create CourseInstructors Table:

CREATE TABLE CourseInstructors (
    CourseID NUMBER,
    InstructorID NUMBER,
    PRIMARY KEY (CourseID, InstructorID),
    FOREIGN KEY (CourseID) REFERENCES Courses(CourseID),
    FOREIGN KEY (InstructorID) REFERENCES Instructors(InstructorID)
);

##Specific Queries
#Select Instructor Names in Department ID 1:

SELECT i.Name
FROM Instructors i
WHERE i.DepartmentID = 1;


#Select Course Titles in Department ID 1:

SELECT c.Title
FROM Courses c
WHERE c.DepartmentID = 1;


#Select Grade for a Specific Student and Course:

SELECT e.Grade
FROM Enrolments e
WHERE e.StudentID = 3 AND e.CourseID = 3;


#Select Course Titles Taught by a Specific Instructor:

SELECT c.Title
FROM Courses c
JOIN CourseInstructors ci ON c.CourseID = ci.CourseID
WHERE ci.InstructorID = 2;


#Select Names and Emails of Students Enrolled in a Specific Course:

SELECT s.Name, s.Email
FROM Students s
JOIN Enrolments e ON s.StudentID = e.StudentID
WHERE e.CourseID = 3;
